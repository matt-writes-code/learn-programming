SQL

Structured Query Language — the universal interface for relational databases. Used in PostgreSQL, MySQL, SQLite, SQL Server, and BigQuery with minor dialect differences.

Declarative

Describe what you want, not how to get it.

ACID

Atomicity, Consistency, Isolation, Durability.

Relational

Data in tables; relationships via keys.

Set-based

Operate on entire result sets at once.

Convention: SQL keywords are commonly written in UPPERCASE; identifiers in lowercase. Not required but widely followed.

SELECT & FROM

select.sql 
-- Select all columns
SELECT * FROM employees;

-- Select specific columns
SELECT first_name, last_name, salary
FROM employees;

-- Column aliases
SELECT
    first_name AS "First Name",
    salary * 12 AS annual_salary
FROM employees;

-- Distinct values
SELECT DISTINCT department_id FROM employees;

-- Expressions in SELECT
SELECT
    name,
    price,
    price * 0.9         AS discounted,
    UPPER(name)         AS name_upper,
    CURRENT_DATE        AS today
FROM products;

WHERE & Filtering

where.sql 
-- Comparison operators
SELECT * FROM employees WHERE salary > 50000;
SELECT * FROM employees WHERE department_id != 5;

-- AND / OR / NOT
SELECT * FROM employees
WHERE salary > 40000
  AND department_id = 2
  AND NOT active = FALSE;

-- BETWEEN (inclusive)
WHERE salary BETWEEN 40000 AND 80000

-- IN / NOT IN
WHERE department_id IN (1, 3, 5)

-- LIKE — pattern matching
WHERE last_name LIKE 'Sm%'        -- starts with "Sm"
WHERE email LIKE '%@gmail.com'   -- ends with
WHERE code LIKE 'A_C'            -- _ = any single char

-- IS NULL / IS NOT NULL
WHERE manager_id IS NULL
WHERE phone IS NOT NULL

-- CASE expression
SELECT
    name,
    CASE
        WHEN salary >= 80000 THEN 'Senior'
        WHEN salary >= 50000 THEN 'Mid'
        ELSE 'Junior'
    END AS level
FROM employees;

ORDER BY & LIMIT

order_limit.sql 
-- Sort ascending (default) / descending
SELECT * FROM products
ORDER BY price ASC;

SELECT * FROM products
ORDER BY price DESC;

-- Multiple columns
SELECT * FROM employees
ORDER BY department_id ASC, salary DESC;

-- LIMIT rows returned
SELECT * FROM products
ORDER BY price DESC
LIMIT 10;

-- OFFSET — for pagination
SELECT * FROM products
ORDER BY id
LIMIT 20 OFFSET 40;   -- page 3, 20 per page

-- NULL ordering
ORDER BY column NULLS LAST

JOINs

JOINs combine rows from two or more tables based on a related column.

joins.sql 
-- INNER JOIN — only matching rows
SELECT e.name, d.department_name
FROM employees e
INNER JOIN departments d ON e.department_id = d.id;

-- LEFT JOIN — all left rows + matches from right (NULL if none)
SELECT e.name, d.department_name
FROM employees e
LEFT JOIN departments d ON e.department_id = d.id;

-- RIGHT JOIN — all right rows + matches from left
SELECT e.name, d.department_name
FROM employees e
RIGHT JOIN departments d ON e.department_id = d.id;

-- FULL OUTER JOIN — all rows from both sides
SELECT e.name, d.department_name
FROM employees e
FULL OUTER JOIN departments d ON e.department_id = d.id;

-- CROSS JOIN — every combination (Cartesian product)
SELECT * FROM sizes CROSS JOIN colors;

-- Self join
SELECT e.name AS employee, m.name AS manager
FROM employees e
JOIN employees m ON e.manager_id = m.id;

-- Multi-table join
SELECT o.id, c.name, p.product_name, oi.quantity
FROM orders o
JOIN customers c    ON o.customer_id = c.id
JOIN order_items oi ON o.id = oi.order_id
JOIN products p     ON oi.product_id = p.id;

Aggregate Functions

Function Description
COUNT(*) Number of rows
COUNT(col) Non-NULL values in column
SUM(col) Sum of values
AVG(col) Average (ignores NULLs)
MIN(col) Smallest value
MAX(col) Largest value
STRING_AGG(col, sep) Concatenate strings (PostgreSQL)
aggregates.sql 
SELECT
    COUNT(*) AS total_employees,
    AVG(salary) AS avg_salary,
    MAX(salary) AS highest_salary,
    MIN(salary) AS lowest_salary,
    SUM(salary) AS total_payroll
FROM employees;

GROUP BY & HAVING

group_by.sql 
-- GROUP BY — aggregate per group
SELECT
    department_id,
    COUNT(*) AS headcount,
    AVG(salary) AS avg_salary
FROM employees
GROUP BY department_id;

-- HAVING — filter on aggregated results (WHERE runs before GROUP BY)
SELECT
    department_id,
    AVG(salary) AS avg_sal
FROM employees
GROUP BY department_id
HAVING AVG(salary) > 60000
ORDER BY avg_sal DESC;

-- Group by multiple columns
SELECT year, month, SUM(revenue)
FROM sales
GROUP BY year, month
ORDER BY year, month;

-- ROLLUP — subtotals + grand total
SELECT region, product, SUM(sales)
FROM data
GROUP BY ROLLUP(region, product);

Subqueries

subqueries.sql 
-- Scalar subquery (returns single value)
SELECT name, salary
FROM employees
WHERE salary > (SELECT AVG(salary) FROM employees);

-- IN subquery
SELECT name FROM customers
WHERE id IN (
    SELECT DISTINCT customer_id FROM orders
    WHERE total > 1000
);

-- EXISTS — efficient membership test
SELECT name FROM customers c
WHERE EXISTS (
    SELECT 1 FROM orders o
    WHERE o.customer_id = c.id
);

-- Derived table (subquery as FROM)
SELECT dept, avg_salary
FROM (
    SELECT department_id AS dept, AVG(salary) AS avg_salary
    FROM employees
    GROUP BY department_id
) AS dept_stats
WHERE avg_salary > 60000;

CTEs (Common Table Expressions)

CTEs create named temporary result sets. Cleaner than subqueries and can be recursive.

ctes.sql 
-- Basic CTE
WITH dept_stats AS (
    SELECT department_id, AVG(salary) AS avg_sal
    FROM employees
    GROUP BY department_id
)
SELECT e.name, e.salary, d.avg_sal
FROM employees e
JOIN dept_stats d ON e.department_id = d.department_id
WHERE e.salary > d.avg_sal;

-- Multiple CTEs
WITH
top_customers AS (
    SELECT customer_id, SUM(total) AS lifetime_value
    FROM orders
    GROUP BY customer_id
    HAVING SUM(total) > 5000
),
recent_orders AS (
    SELECT * FROM orders
    WHERE created_at >= NOW() - INTERVAL '30 days'
)
SELECT c.name, tc.lifetime_value
FROM top_customers tc
JOIN customers c ON tc.customer_id = c.id
WHERE EXISTS (SELECT 1 FROM recent_orders r WHERE r.customer_id = tc.customer_id);

-- Recursive CTE — traverse hierarchy
WITH RECURSIVE org_chart AS (
    SELECT id, name, manager_id, 0 AS level
    FROM employees
    WHERE manager_id IS NULL      -- root
    UNION ALL
    SELECT e.id, e.name, e.manager_id, oc.level + 1
    FROM employees e
    JOIN org_chart oc ON e.manager_id = oc.id
)
SELECT * FROM org_chart ORDER BY level, name;

INSERT / UPDATE / DELETE

dml.sql 
-- INSERT single row
INSERT INTO products (name, price, category_id)
VALUES ('Widget Pro', 29.99, 3);

-- INSERT multiple rows
INSERT INTO products (name, price)
VALUES
    ('Alpha', 10.00),
    ('Beta',  15.00),
    ('Gamma', 20.00);

-- INSERT from SELECT
INSERT INTO archive_orders
SELECT * FROM orders WHERE created_at < '2023-01-01';

-- UPDATE
UPDATE employees
SET salary = salary * 1.1,
    updated_at = NOW()
WHERE department_id = 2
  AND performance_rating = 'exceeds';

-- DELETE
DELETE FROM sessions
WHERE expires_at < NOW();

-- UPSERT (PostgreSQL)
INSERT INTO users (email, name)
VALUES ('a@b.com', 'Alice')
ON CONFLICT (email)
DO UPDATE SET name = EXCLUDED.name;

-- RETURNING — get inserted/updated data back
INSERT INTO orders (customer_id, total)
VALUES (1, 99.99)
RETURNING id, created_at;

DDL & Data Types

ddl.sql 
-- CREATE TABLE
CREATE TABLE users (
    id          SERIAL PRIMARY KEY,      -- auto-increment integer
    email       VARCHAR(255) UNIQUE NOT NULL,
    name        TEXT NOT NULL,
    age         INTEGER CHECK (age >= 0),
    role        TEXT DEFAULT 'user',
    balance     NUMERIC(12, 2) DEFAULT 0,
    active      BOOLEAN DEFAULT TRUE,
    metadata    JSONB,
    created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- Foreign key
CREATE TABLE orders (
    id          SERIAL PRIMARY KEY,
    user_id     INTEGER REFERENCES users(id) ON DELETE CASCADE,
    total       NUMERIC(10, 2) NOT NULL
);

-- Alter table
ALTER TABLE users ADD COLUMN phone TEXT;
ALTER TABLE users DROP COLUMN age;
ALTER TABLE users RENAME COLUMN name TO full_name;
ALTER TABLE users ALTER COLUMN role SET DEFAULT 'viewer';

-- Drop
DROP TABLE IF EXISTS temp_data;
TRUNCATE TABLE sessions;   -- fast delete all rows

Indexes

Indexes speed up reads at the cost of slower writes and extra storage. Add them on columns frequently used in WHERE, JOIN, and ORDER BY.

indexes.sql 
-- B-Tree index (default) — good for =, <, >, BETWEEN, LIKE 'prefix%'
CREATE INDEX idx_users_email ON users (email);

-- Unique index
CREATE UNIQUE INDEX idx_users_email_unique ON users (email);

-- Composite index (column order matters!)
CREATE INDEX idx_orders_user_date ON orders (user_id, created_at);

-- Partial index — index subset of rows
CREATE INDEX idx_active_users ON users (email)
WHERE active = TRUE;

-- GIN index for JSONB / full-text search
CREATE INDEX idx_metadata ON users USING GIN (metadata);

-- EXPLAIN ANALYZE — understand query execution
EXPLAIN ANALYZE
SELECT * FROM users WHERE email = 'test@example.com';

Transactions

transactions.sql 
-- Explicit transaction
BEGIN;

UPDATE accounts SET balance = balance - 100 WHERE id = 1;
UPDATE accounts SET balance = balance + 100 WHERE id = 2;

COMMIT;     -- persist both changes
-- or ROLLBACK; to undo both

-- Savepoints — partial rollback
BEGIN;
INSERT INTO orders ... ;
SAVEPOINT sp1;
INSERT INTO order_items ... ;
ROLLBACK TO sp1;       -- undo items but keep order
COMMIT;

-- Isolation levels (PostgreSQL)
SET TRANSACTION ISOLATION LEVEL REPEATABLE READ;
-- READ COMMITTED (default), REPEATABLE READ, SERIALIZABLE

Window Functions

Window functions perform calculations across rows related to the current row, without collapsing them into groups.

window.sql 
-- ROW_NUMBER, RANK, DENSE_RANK
SELECT
    name, salary, department_id,
    ROW_NUMBER()  OVER (PARTITION BY department_id ORDER BY salary DESC) AS row_num,
    RANK()        OVER (PARTITION BY department_id ORDER BY salary DESC) AS rnk,
    DENSE_RANK()  OVER (PARTITION BY department_id ORDER BY salary DESC) AS dense_rnk
FROM employees;

-- Running total & moving average
SELECT
    date,
    revenue,
    SUM(revenue) OVER (ORDER BY date ROWS UNBOUNDED PRECEDING) AS running_total,
    AVG(revenue) OVER (ORDER BY date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) AS moving_avg_7d
FROM daily_revenue;

-- LAG / LEAD — previous / next row value
SELECT
    date,
    revenue,
    LAG(revenue, 1) OVER (ORDER BY date) AS prev_day,
    revenue - LAG(revenue, 1) OVER (ORDER BY date) AS change
FROM daily_revenue;

-- NTILE — divide into buckets
NTILE(4) OVER (ORDER BY salary) AS quartile

-- Top 1 per group pattern
SELECT * FROM (
    SELECT *, ROW_NUMBER() OVER (PARTITION BY dept ORDER BY salary DESC) AS rn
    FROM employees
) WHERE rn = 1;

Views

views.sql 
-- Regular view — saved query, no stored data
CREATE OR REPLACE VIEW active_users AS
SELECT id, name, email
FROM users
WHERE active = TRUE;

SELECT * FROM active_users;   -- use like a table

-- Materialized view — stores results for performance
CREATE MATERIALIZED VIEW monthly_sales AS
SELECT
    DATE_TRUNC('month', created_at) AS month,
    SUM(total) AS revenue
FROM orders
GROUP BY 1;

-- Refresh materialized view
REFRESH MATERIALIZED VIEW monthly_sales;

DROP VIEW IF EXISTS active_users;

JSON in SQL

json.sql 
-- PostgreSQL JSONB operators
-- ->  returns JSON,  ->> returns text
SELECT
    metadata->'address'           AS address_json,
    metadata->>'city'            AS city_text,
    metadata#>>'{address,zip}'  AS zip
FROM users;

-- Filter on JSON field
SELECT * FROM users
WHERE metadata->>'plan' = 'premium';

-- JSON containment @>
WHERE metadata @> '{"role": "admin"}'

-- Build JSON
SELECT json_build_object(
    'name', name,
    'email', email
) AS user_json
FROM users;

-- Aggregate to JSON array
SELECT json_agg(name) FROM users;

-- Update a JSON field
UPDATE users
SET metadata = metadata || '{"plan": "premium"}'
WHERE id = 1;
