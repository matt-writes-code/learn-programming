React

A JavaScript library for building user interfaces via composable components. Developed by Meta. Uses a virtual DOM for efficient UI updates.

Component-based

Encapsulated components that manage their own state.

Declarative

Describe what the UI should look like, React handles updates.

Unidirectional

Data flows down. Events bubble up.

Hooks

Functions to use state and lifecycle in function components.

JSX

JSX is a syntax extension that looks like HTML inside JavaScript. It compiles to React.createElement() calls.

jsx.jsx 
// JSX — HTML-like syntax in JS
const element = <h1 className="title">Hello World</h1>;

// Embed expressions with { }
const name = "Alice";
const el = <p>Hello, {name}!</p>;

// All tags must be closed
<img src="logo.png" alt="Logo" />
<br />

// className instead of class, htmlFor instead of for
<label htmlFor="email">Email</label>
<input id="email" className="input" />

// Style as object
const style = { color: 'red', fontSize: '16px' };
<div style={style}>Styled</div>

// JSX must have one root element — use Fragment to avoid extra DOM
return (
  <>
    <h1>Title</h1>
    <p>Body</p>
  </>
);

State (useState)

useState.jsx 
import { useState } from 'react';

function Counter() {
  // [value, setter] — setter triggers re-render
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>+</button>
      <button onClick={() => setCount(c => c - 1)}>-</button>
    </div>
  );
}

// Object state — must spread to preserve other fields
const [user, setUser] = useState({ name: '', age: 0 });
setUser(u => ({ ...u, name: 'Alice' }));

// Lazy initializer — runs once on mount
const [data] = useState(() => JSON.parse(localStorage.getItem('data')));

// Never mutate state directly:
// ❌  count++
// ✅  setCount(c => c + 1)

Effects (useEffect)

useEffect.jsx 
import { useState, useEffect } from 'react';

function Profile({ userId }) {
  const [user, setUser] = useState(null);

  // Runs after render when userId changes
  useEffect(() => {
    let cancelled = false;

    async function load() {
      const data = await fetchUser(userId);
      if (!cancelled) setUser(data);
    }

    load();
    return () => { cancelled = true; };  // cleanup on unmount/re-run
  }, [userId]);  // dependency array

  if (!user) return <p>Loading…</p>;
  return <p>{user.name}</p>;
}

// Dependency array behaviour:
// useEffect(fn)          — runs after every render
// useEffect(fn, [])      — runs once on mount
// useEffect(fn, [a, b])  — runs when a or b changes

// Event listener cleanup pattern
useEffect(() => {
  const handler = (e) => setKey(e.key);
  window.addEventListener('keydown', handler);
  return () => window.removeEventListener('keydown', handler);
}, []);

Lists & Keys

lists.jsx 
// Render a list — every item needs a unique key
function TodoList({ todos }) {
  return (
    <ul>
      {todos.map(todo => (
        <li key={todo.id}>
          {todo.text}
        </li>
      ))}
    </ul>
  );
}

// Key rules:
// ✅  Use stable, unique IDs (todo.id, user.email)
// ❌  Don't use array index if list can be reordered/filtered
// ❌  Don't use random keys — defeats reconciliation

// Filtering & transforming before rendering
const activeTodos = todos
  .filter(t => !t.done)
  .map(t => <TodoItem key={t.id} todo={t} />);

Forms

forms.jsx 
// Controlled inputs — React owns the value
function LoginForm() {
  const [form, setForm] = useState({ email: '', password: '' });

  const handleChange = (e) =>
    setForm(f => ({ ...f, [e.target.name]: e.target.value }));

  const handleSubmit = (e) => {
    e.preventDefault();
    console.log(form);
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        name="email"
        type="email"
        value={form.email}
        onChange={handleChange}
      />
      <input
        name="password"
        type="password"
        value={form.password}
        onChange={handleChange}
      />
      <button type="submit">Login</button>
    </form>
  );
}

Context API

Context shares data across the component tree without prop-drilling.

context.jsx 
import { createContext, useContext, useState } from 'react';

// 1. Create context with default value
const ThemeContext = createContext('light');

// 2. Provide value high in the tree
function App() {
  const [theme, setTheme] = useState('dark');
  return (
    <ThemeContext.Provider value={theme}>
      <Page />
    </ThemeContext.Provider>
  );
}

// 3. Consume anywhere in the tree
function Page() {
  const theme = useContext(ThemeContext);
  return <div className={theme}>Page content</div>;
}

// Custom hook wrapper (idiomatic)
function useTheme() {
  const ctx = useContext(ThemeContext);
  if (!ctx) throw new Error('useTheme must be inside ThemeProvider');
  return ctx;
}

useReducer

Better than useState when state logic is complex or next state depends on previous.

useReducer.jsx 
import { useReducer } from 'react';

const reducer = (state, action) => {
  switch (action.type) {
    case 'increment': return { count: state.count + 1 };
    case 'decrement': return { count: state.count - 1 };
    case 'reset':     return { count: 0 };
    default: return state;
  }
};

function Counter() {
  const [state, dispatch] = useReducer(reducer, { count: 0 });
  return (
    <>
      <p>{state.count}</p>
      <button onClick={() => dispatch({ type: 'increment' })}>+</button>
      <button onClick={() => dispatch({ type: 'reset' })}>Reset</button>
    </>
  );
}

useRef

useRef.jsx 
import { useRef, useEffect } from 'react';

// DOM reference
function FocusInput() {
  const inputRef = useRef(null);

  useEffect(() => {
    inputRef.current.focus();   // focus on mount
  }, []);

  return <input ref={inputRef} />;
}

// Mutable value that doesn't trigger re-render
function Timer() {
  const intervalRef = useRef(null);

  const start = () => {
    intervalRef.current = setInterval(() => tick(), 1000);
  };

  const stop = () => clearInterval(intervalRef.current);

  return <><button onClick={start}>Start</button><button onClick={stop}>Stop</button></>;
}

// Previous value pattern
function usePrevious(value) {
  const ref = useRef();
  useEffect(() => { ref.current = value; });
  return ref.current;
}

useMemo & useCallback

memo.jsx 
import { useMemo, useCallback, memo } from 'react';

// useMemo — cache expensive computation
const sortedItems = useMemo(
  () => [...items].sort((a, b) => a.price - b.price),
  [items]   // recompute only when items changes
);

// useCallback — stable function reference (for prop equality)
const handleDelete = useCallback(
  (id) => setItems(prev => prev.filter(i => i.id !== id)),
  []
);

// React.memo — skip re-render if props unchanged
const ItemRow = memo(function ItemRow({ item, onDelete }) {
  return (
    <div>
      {item.name}
      <button onClick={() => onDelete(item.id)}>×</button>
    </div>
  );
});

// Rule: only add memoization when profiling shows a real problem.
// Premature optimization adds complexity for no gain.

Custom Hooks

Extract reusable stateful logic into functions starting with use.

customHooks.js 
// useFetch — generic data fetching
function useFetch(url) {
  const [data,    setData]    = useState(null);
  const [loading, setLoading] = useState(true);
  const [error,   setError]   = useState(null);

  useEffect(() => {
    let cancelled = false;
    fetch(url)
      .then(r => r.json())
      .then(d => { if (!cancelled) setData(d); })
      .catch(e => { if (!cancelled) setError(e); })
      .finally(() => { if (!cancelled) setLoading(false); });
    return () => { cancelled = true; };
  }, [url]);

  return { data, loading, error };
}

// useLocalStorage
function useLocalStorage(key, initial) {
  const [value, setValue] = useState(
    () => JSON.parse(localStorage.getItem(key)) ?? initial
  );
  const set = v => { setValue(v); localStorage.setItem(key, JSON.stringify(v)); };
  return [value, set];
}

// useDebounce
function useDebounce(value, delay = 300) {
  const [debounced, setDebounced] = useState(value);
  useEffect(() => {
    const t = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(t);
  }, [value, delay]);
  return debounced;
}

patterns.jsx 
// Compound components — shared implicit state
function Tabs({ children, defaultTab }) {
  const [active, setActive] = useState(defaultTab);
  return (
    <TabContext.Provider value={{ active, setActive }}>
      {children}
    </TabContext.Provider>
  );
}

// Render props — pass rendering logic as a prop
<DataProvider render={(data) => <Chart data={data} />} />

// Higher-Order Component (HOC)
function withAuth(Component) {
  return function AuthGuard(props) {
    const { user } = useAuth();
    if (!user) return <Redirect to="/login" />;
    return <Component {...props} user={user} />;
  };
}
