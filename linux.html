Linux

An open-source Unix-like OS kernel powering servers, cloud infrastructure, Android, and most developer machines. Understanding the shell unlocks immense productivity and control.

Everything is a file

Devices, pipes, and sockets are all accessed via file paths.

Small tools

Many focused utilities, composed with pipes and redirects.

Multi-user

Strong permission model isolates users and processes.

Distributions

Ubuntu/Debian, RHEL/Fedora, Arch, Alpine, etc.

File System Hierarchy

Path Contents
/ Root — top of the tree
/bin, /usr/bin Essential user commands
/sbin, /usr/sbin System admin commands
/etc System-wide config files
/home User home directories
/root Root user's home
/tmp Temporary files (cleared on boot)
/var Variable data: logs, caches, spools
/proc Virtual FS: kernel & process info
/dev Device files (disks, terminals)
/opt Optional third-party software
/usr/local Locally installed software

File Operations

files.sh 
# Create
touch file.txt                # create empty file / update timestamp
mkdir mydir                   # create directory

# Copy
cp file.txt backup.txt

cp -p file.txt dest/          # preserve permissions/timestamps

Permissions

Format: -rwxrwxrwx — file type, then owner/group/other permissions (r=4, w=2, x=1).
permissions.sh 
# View permissions
ls -l file.txt
# -rw-r--r-- 1 alice staff 1234 Jan 1 12:00 file.txt
#  │││└─┤└─┤ │ └───┘ └───┘
#  │││  │  │ │ owner  group
#  │││  │  └─ other (r--)  = 4
#  │││  └──── group (r--)  = 4
#  │└┴──────── owner (rw-) = 6
#  └─────────── - = regular file, d = dir, l = symlink

# chmod — change permissions
chmod 644 file.txt            # owner rw, group r, other r
chmod 755 script.sh           # owner rwx, group rx, other rx
chmod 600 ~/.ssh/id_rsa       # owner rw only (private key)
chmod u+x script.sh           # add execute for owner
chmod go-w file.txt           # remove write from group & other
chmod -R 755 mydir/           # recursive

# chown — change ownership
chown alice file.txt
chown alice:developers file.txt
chown -R alice:staff mydir/

# Special permissions
chmod u+s script              # setuid — runs as file owner
chmod g+s dir/                # setgid — new files inherit group
chmod +t /tmp                 # sticky bit — only owner can delete

Users & Groups

# Switch user

# User management (root/sudo required)
useradd -m -s /bin/bash alice  # create user with home & shell

# Group management

gpasswd -d alice developers    # remove alice from group

# User info files
cat /etc/passwd                # user accounts
cat /etc/group                 # group definitions
last                           # login history
who                            # currently logged in users

Pipes & Redirection

pipes.sh 
# Redirection
command > output.txt          # stdout → file (overwrite)
command >> output.txt         # stdout → file (append)
command 2> errors.txt         # stderr → file
command 2>&1                  # stderr → stdout
command > out.txt 2>&1        # both to file
command < input.txt           # stdin from file

# /dev/null — discard output
command >/dev/null 2>&1

# Pipe — stdout of left → stdin of right
ls -l | grep ".txt"
cat access.log | sort | uniq -c | sort -rn | head -20
ps aux | grep nginx | awk '{print $2}'

# tee — write to file AND pass to next command
command | tee output.txt | grep "error"

# xargs — build commands from stdin
find . -name "*.log" | xargs rm
cat urls.txt | xargs -I{} curl -O {}

# Process substitution
diff <(ls dir1) <(ls dir2)

Text Processing

text.sh 
# sort
sort file.txt
sort -r file.txt              # reverse
sort -n numbers.txt           # numeric sort
sort -k2 file.txt             # sort by 2nd field

# uniq — remove adjacent duplicates
sort file.txt | uniq
sort file.txt | uniq -c       # count occurrences
sort file.txt | uniq -d       # only duplicates

# cut — extract columns
cut -d: -f1,3 /etc/passwd     # fields 1 & 3, colon delimiter
cut -c1-10 file.txt           # chars 1-10

# awk — field-based text processing
awk '{print $1, $3}' file.txt            # fields 1 and 3
awk -F: '{print $1}' /etc/passwd         # custom delimiter
awk '$3 > 1000 {print $1}' file.txt      # conditional
awk '{sum += $2} END {print sum}' data   # accumulator

# sed — stream editor
sed 's/old/new/' file.txt                # replace first occurrence per line
sed 's/old/new/g' file.txt              # replace all
sed -i 's/http/https/g' config.txt      # in-place edit
sed '3,5d' file.txt                     # delete lines 3-5
sed -n '10,20p' file.txt                # print lines 10-20

# tr — translate/delete characters
echo "HELLO" | tr 'A-Z' 'a-z'
echo "hello world" | tr ' ' '_'
cat file.txt | tr -d '\r'               # remove Windows line endings

Shell Scripting

backup.sh 
#!/usr/bin/env bash
# Shebang: specify interpreter
set -euo pipefail             # exit on error, unset vars, pipe errors

# Variables
NAME="World"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="/backups/${TIMESTAMP}"

# String interpolation
echo "Hello, ${NAME}!"

# Conditionals
if [ -d "$BACKUP_DIR" ]; then
  echo "Directory exists"
elif [ -f "$BACKUP_DIR" ]; then
  echo "Is a file"
else
  mkdir -p "$BACKUP_DIR"
fi

# Test operators
# [ -e file ]  exists   [ -f file ]  regular file
# [ -d dir  ]  directory [ -r file ] readable
# [ -z str  ]  empty     [ -n str  ] non-empty
# [ a == b  ]  equal     [ a != b  ] not equal
# [ a -lt b ]  less than [ a -gt b ] greater than

# Loops
for file in /var/log/*.log; do
  echo "Processing: $file"
done

while [ $count -lt 10 ]; do
  count=$((count + 1))
done

# Functions
log() {
  echo "[$(date '+%T')] $*" | tee -a /var/log/myscript.log
}

backup_dir() {
  local src="$1"
  local dest="$2"
  tar czf "${dest}.tar.gz" "$src"
  log "Backed up $src to ${dest}.tar.gz"
}

log "Starting backup"
backup_dir /etc /backups/etc

# Exit codes
command && echo "success" || echo "failed"
command; echo "Exit code: $?"

Cron Jobs

Schedule recurring tasks with cron. Fields: minute hour day-of-month month day-of-week.

crontab 
# Edit crontab
crontab -e
crontab -l                    # list current jobs
crontab -r                    # remove all jobs

# Format:  MIN HOUR DOM MON DOW COMMAND
# MIN  0-59    HOUR 0-23
# DOM  1-31    MON  1-12
# DOW  0-7 (0 and 7 = Sunday)

# Examples
0 * * * *      /script.sh         # every hour at :00
*/15 * * * *   /health-check.sh   # every 15 minutes
0 2 * * *      /backup.sh         # daily at 2:00 AM
0 2 * * 0      /weekly.sh         # every Sunday at 2 AM
0 0 1 * *      /monthly.sh        # 1st of every month
@reboot         /startup.sh        # on system startup

# Redirect output
0 3 * * * /backup.sh >> /var/log/backup.log 2>&1

# systemd timers (modern alternative)
systemctl list-timers

Vim

Vim is a modal text editor. It has modes: Normal (default), Insert (i), Visual (v), Command (:).

vim cheatsheet 
# Open / exit
vim file.txt                  # open file
:w                            # save
:q                            # quit
:wq  or  :x                   # save + quit
:q!                           # quit without saving
ZZ                            # save + quit (normal mode)

# Mode switching
i                             # insert before cursor
a                             # insert after cursor
o                             # new line below, insert
O                             # new line above, insert
v                             # visual (character) mode
V                             # visual line mode
Ctrl+v                        # visual block mode
Esc                           # return to normal mode

# Navigation (normal mode)
h j k l                       # left down up right
w / b                         # next / prev word
0 / $                         # start / end of line
gg / G                        # first / last line
:42                           # go to line 42
Ctrl+f / Ctrl+b               # page down / up

# Edit (normal mode)
dd                            # delete line
yy                            # yank (copy) line
p                             # paste after
u                             # undo
Ctrl+r                        # redo
x                             # delete character
cw                            # change word

# Search & replace
/pattern                      # search forward
n / N                         # next / prev match
:%s/old/new/g                 # replace all in file
:5,10s/old/new/g              # replace in lines 5-10

SSH

ssh.sh 
# Connect
ssh user@hostname
ssh user@192.168.1.100
ssh -p 2222 user@host          # custom port
ssh -i ~/.ssh/id_rsa user@host # specific key

# Key management
ssh-keygen -t ed25519 -C "alice@example.com"
# creates ~/.ssh/id_ed25519 and ~/.ssh/id_ed25519.pub

ssh-copy-id user@host          # copy public key to server
# or manually: cat ~/.ssh/id_ed25519.pub | ssh user@host \
#   "mkdir -p ~/.ssh && cat >> ~/.ssh/authorized_keys"

# SSH config (~/.ssh/config)
Host myserver
    HostName 192.168.1.100
    User alice
    IdentityFile ~/.ssh/id_ed25519
    Port 2222

ssh myserver                   # uses config above

# File transfer
scp file.txt user@host:/path/
scp -r dir/ user@host:/remote/
rsync -avz src/ user@host:dest/ # fast, incremental sync

# Tunnels
ssh -L 5432:db-host:5432 bastion  # local port forwarding
ssh -R 8080:localhost:80 server   # remote port forwarding
ssh -D 1080 user@host             # SOCKS proxy

# Persistent sessions
tmux new -s main
tmux attach -t main
screen -S session
