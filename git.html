Git

A distributed version control system that tracks changes, enables collaboration, and lets you roll back to any point in history.

Distributed

Every clone is a full repository with complete history.

Snapshots

Git stores file snapshots, not diffs. Fast and efficient.

Branching

Branches are cheap pointers to commits — create freely.

SHA-1 integrity

Every object is identified by a content hash.

Three areas: Working directory (your files) → Staging area (index, what's in the next commit) → Repository (.git/objects, the history).

Setup & Config

bash 
# Identity (required before first commit)
git config --global user.name  "Alice Smith"
git config --global user.email "alice@example.com"

# Default branch name
git config --global init.defaultBranch main

# Editor for commit messages
git config --global core.editor "code --wait"

# Aliases
git config --global alias.st status
git config --global alias.co checkout
git config --global alias.br branch
git config --global alias.lg "log --oneline --graph --decorate --all"

# Scope: --global (user), --local (repo), --system (machine)
git config --list                # show all settings

Staging & Committing

bash 
# Check status
git status
git status -s                    # short format

# Stage files
git add file.txt                 # specific file
git add src/                     # directory
git add .                        # all changes in cwd
git add -p                       # interactive: stage hunks

# Unstage (without losing changes)
git restore --staged file.txt
git reset HEAD file.txt          # older syntax

# Commit
git commit -m "feat: add login page"
git commit                       # opens editor
git commit -am "fix: typo"       # stage tracked + commit

# Amend last commit (before pushing!)
git commit --amend -m "corrected message"
git commit --amend --no-edit     # keep message, add staged files

Branching

bash 
# List branches
git branch                       # local
git branch -a                    # all (including remote)
git branch -v                    # with last commit

# Create branch
git branch feature/login
git checkout -b feature/login    # create + switch
git switch -c feature/login      # modern syntax

# Switch branch
git checkout main
git switch main                  # modern syntax

# Rename branch
git branch -m old-name new-name
git branch -m new-name           # rename current

# Delete branch
git branch -d feature/login      # safe (merged only)
git branch -D feature/login      # force delete

# Create branch from specific commit
git checkout -b hotfix/bug abc1234

Merging

bash 
# Merge feature branch into main
git switch main
git merge feature/login

# No-fast-forward — always creates merge commit
git merge --no-ff feature/login

# Squash merge — squashes all commits into one staged change
git merge --squash feature/login
git commit -m "feat: login page"

# Resolve conflicts:
#  1. Open conflicted files — look for <<<<, ====, >>>>
#  2. Edit to desired state
#  3. git add <resolved-file>
#  4. git commit

# Abort merge
git merge --abort

# Check for merge conflicts before merging
git merge --no-commit --no-ff feature/login

Rebase

Golden rule: Never rebase commits that have been pushed to a shared remote branch. Rebase rewrites history.
bash 
# Rebase current branch onto main
git switch feature/login
git rebase main

# Interactive rebase — edit, squash, reorder commits
git rebase -i HEAD~3             # last 3 commits
git rebase -i main               # all commits since branching off main

# Interactive options:
#  pick  p  — use commit as-is
#  reword r  — change commit message
#  edit  e  — pause to amend commit
#  squash s  — combine with previous commit
#  fixup f  — squash, discard this commit's message
#  drop  d  — remove commit

# Continue / abort rebase
git rebase --continue
git rebase --abort

# Pull with rebase (cleaner history)
git pull --rebase origin main

Remote Repositories

bash 
# Manage remotes
git remote -v                              # list remotes
git remote add origin https://github.com/user/repo.git
git remote rename origin upstream
git remote remove upstream

# Fetch — download without merging
git fetch origin                           # all branches
git fetch origin main                      # specific branch

# Pull = fetch + merge (or rebase)
git pull origin main
git pull --rebase origin main

# Push
git push origin main
git push -u origin feature/login           # set upstream tracking
git push                                   # after -u, shorthand
git push --force-with-lease                # safer force push

# Delete remote branch
git push origin --delete feature/login

# Track remote branch
git branch --set-upstream-to=origin/main main

Stash

Save uncommitted changes temporarily without committing them.

bash 
# Stash current changes
git stash
git stash push -m "WIP: half-done feature"
git stash push -u                          # include untracked files

# List stashes
git stash list

# Apply stash
git stash pop                              # apply + remove from stash
git stash apply stash@{2}                 # apply specific, keep in stash

# View stash contents
git stash show -p stash@{0}

# Drop / clear stashes
git stash drop stash@{0}
git stash clear                            # remove all stashes

# Create branch from stash
git stash branch feature/new stash@{0}

Log & Diff

bash 
# Log
git log
git log --oneline                          # compact
git log --oneline --graph --decorate --all # visual tree
git log --author="Alice"
git log --since="2024-01-01" --until="2024-12-31"
git log -p                                 # show patch/diff
git log --stat                             # files changed
git log -- path/to/file                    # history of a file

# Diff
git diff                                   # working vs staged
git diff --staged                          # staged vs last commit
git diff main..feature/login               # between branches
git diff abc1234..def5678                  # between commits
git diff HEAD~3                            # vs 3 commits ago

# Show a commit
git show abc1234
git show HEAD:path/to/file                 # file at a commit

# Who changed a line?
git blame path/to/file

Undoing Changes

Command Effect Safe?
git restore file Discard working dir changes ⚠️ Destructive
git restore --staged file Unstage file ✅ Safe
git revert <commit> New commit that undoes a commit ✅ Safe
git reset --soft HEAD~1 Undo commit, keep staged ✅ Local only
git reset --mixed HEAD~1 Undo commit, keep working dir ✅ Local only
git reset --hard HEAD~1 Undo commit, discard all changes ⚠️ Destructive
bash 
# Discard working directory changes
git restore file.txt                       # single file
git restore .                              # all files

# Revert (adds new commit undoing the old one — safe for shared history)
git revert HEAD                            # undo last commit
git revert abc1234                         # undo specific commit

# Reset (rewrites history — use locally only)
git reset --soft HEAD~1   # undo commit, staged changes preserved
git reset --mixed HEAD~1  # undo commit, changes in working dir
git reset --hard HEAD~1   # undo commit, all changes gone

# Recover a dropped commit via reflog
git reflog
git checkout -b recover abc1234

Tags

bash 
# Annotated tag (preferred — includes message + author)
git tag -a v1.0.0 -m "Release v1.0.0"

# Lightweight tag
git tag v1.0.0-rc1

# Tag a specific commit
git tag -a v0.9.0 abc1234 -m "Beta release"

# List & show
git tag
git tag -l "v1.*"
git show v1.0.0

# Push tags
git push origin v1.0.0                    # one tag
git push origin --tags                    # all tags

# Delete tag
git tag -d v1.0.0                         # local
git push origin --delete v1.0.0           # remote

Cherry-pick

Apply the changes from a specific commit to the current branch.

bash 
# Apply a single commit
git cherry-pick abc1234

# Apply a range of commits
git cherry-pick abc1234..def5678

# Cherry-pick without committing
git cherry-pick --no-commit abc1234

# Handle conflicts then continue
git cherry-pick --continue
git cherry-pick --abort

Workflows

feature-branch.sh 
# 1. Always start from up-to-date main
git switch main
git pull origin main

# 2. Create feature branch
git switch -c feature/user-auth

# 3. Work, commit often
git add -p
git commit -m "feat: add JWT validation"

# 4. Keep up to date with main
git fetch origin
git rebase origin/main

# 5. Push & open pull request
git push -u origin feature/user-auth

# 6. After PR merged, clean up
git switch main
git pull origin main
git branch -d feature/user-auth
gitflow.sh 
# Branches:
#  main     — production releases, tagged
#  develop  — integration branch
#  feature/* — new features
#  release/* — release prep
#  hotfix/*  — emergency production fixes

# New feature
git switch -c feature/payments develop

# Finish feature
git switch develop
git merge --no-ff feature/payments
git branch -d feature/payments

# Hotfix on production
git switch -c hotfix/1.0.1 main
git commit -m "fix: critical security patch"
git switch main  && git merge --no-ff hotfix/1.0.1
git tag v1.0.1
git switch develop && git merge --no-ff hotfix/1.0.1
trunk-based.sh 
# Trunk-based: everyone commits to main frequently
# Feature flags hide incomplete work
# Short-lived branches (< 2 days)

git switch main
git pull --rebase origin main

# Very short-lived feature branch
git switch -c fix/typo-in-header
git commit -m "fix: correct heading text"
git push origin fix/typo-in-header

# Merge same day via PR or direct push
git switch main
git merge fix/typo-in-header
git push origin main

Advanced

bash 
# bisect — binary search for bad commit
git bisect start
git bisect bad                             # current is bad
git bisect good v1.0.0                    # this was good
# Git checks out midpoints — mark each good/bad
git bisect good / git bisect bad
git bisect reset                           # done

# worktree — multiple branches checked out simultaneously
git worktree add ../hotfix hotfix/critical-bug

# submodule — embed another repo
git submodule add https://github.com/user/lib.git lib
git submodule update --init --recursive

# sparse checkout — only fetch subset of files
git sparse-checkout set src/ docs/

# clean — remove untracked files
git clean -n                               # dry run
git clean -fd                              # force delete files + dirs
