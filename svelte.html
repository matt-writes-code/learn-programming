Svelte

A compiler-based UI framework — no virtual DOM. Svelte compiles your components to vanilla JS at build time, producing tiny, fast bundles with minimal boilerplate.

No Virtual DOM

Svelte surgically updates the real DOM — no diffing overhead.

Compile-time

Components compile away. Zero framework runtime in production.

Truly Reactive

Reactivity is built into the language via assignments.

Less Code

Significantly fewer lines than equivalent React or Vue code.

Component Anatomy

Every .svelte file is a self-contained component with three optional sections.

MyComponent.svelte 
<script>
  // Component logic — runs once on creation
  let name = 'World';
  let count = 0;

  function increment() {
    count += 1;  // assignment triggers reactivity
  }
</script>

<!-- Template — HTML + Svelte directives -->
<h1>Hello, {name}!</h1>
<button on:click={increment}>Count: {count}</button>

<style>
  /* Scoped by default — won't leak to other components */
  h1 {
    color: coral;
    font-size: 2rem;
  }
</style>

Reactivity

reactivity.svelte 
<script>
  let width  = 10;
  let height = 5;

  // Reactive declaration — recomputes when dependencies change
  $: area = width * height;
  $: perimeter = 2 * (width + height);

  // Reactive statement — runs as a side-effect
  $: console.log(`Area changed to ${area}`);

  // Reactive block
  $: {
    if (area > 100) {
      alert('Very large!');
    }
  }

  // Arrays and objects: must reassign to trigger reactivity
  let items = ['a', 'b'];

  // ❌ items.push('c') — Svelte can't detect this
  // ✅ items = [...items, 'c']
  function addItem() {
    items = [...items, 'c'];
  }

  // Shorthand for object mutation
  let user = { name: 'Alice', age: 30 };
  function birthday() {
    user.age += 1;
    user = user;  // triggers reactivity (Svelte 4)
  }
</script>

<p>Width: <input type="number" bind:value={width}></p>
<p>Area: {area}</p>

Template Logic

logic.svelte 
<!-- if / else if / else -->
{#if status === 'loading'}
  <Spinner />
{:else if status === 'error'}
  <p class="error">Something went wrong</p>
{:else}
  <Content />
{/if}

<!-- each — iterate arrays -->
{#each items as item (item.id)}
  <div>{item.name}</div>
{:else}
  <p>No items</p>
{/each}

<!-- each with index -->
{#each items as item, i (item.id)}
  <p>{i + 1}. {item.name}</p>
{/each}

<!-- await — handle promises directly in markup -->
{#await fetchUser(id)}
  <p>Loading…</p>
{:then user}
  <p>{user.name}</p>
{:catch error}
  <p>Error: {error.message}</p>
{/await}

<!-- key block — force remount when key changes -->
{#key userId}
  <Profile id={userId} />
{/key}

Events

events.svelte 
<script>
  import { createEventDispatcher } from 'svelte';
  const dispatch = createEventDispatcher();

  function handleSelect(item) {
    // Dispatch custom event to parent
    dispatch('select', { item });
  }
</script>

<!-- DOM events -->
<button on:click={handleClick}>Click</button>

<!-- Inline handler -->
<button on:click={() => count++}>+</button>

<!-- Event modifiers -->
<button on:click|once={fn}>Once</button>
<form   on:submit|preventDefault={submit}>...</form>
<div    on:click|stopPropagation={fn}>...</div>
<!-- Other modifiers: passive, capture, self, trusted -->

<!-- Event forwarding (pass DOM event to parent) -->
<button on:click>Forward click</button>

<!-- In parent: listen to custom component event -->
<MyList on:select={(e) => selected = e.detail.item} />

Lifecycle

lifecycle.svelte 
<script>
  import {
    onMount,
    onDestroy,
    beforeUpdate,
    afterUpdate,
    tick
  } from 'svelte';

  // onMount — runs after first render (client only)
  onMount(async () => {
    const data = await fetchData();
    // Return a cleanup function
    return () => cleanup();
  });

  // onDestroy — runs when component is removed
  onDestroy(() => {
    clearInterval(timer);
  });

  // beforeUpdate / afterUpdate — before/after DOM updates
  beforeUpdate(() => { /* capture scroll position */ });
  afterUpdate(() =>  { /* restore scroll position */ });

  // tick — wait for pending DOM updates
  async function handleInput() {
    value = 'new';
    await tick();
    inputEl.select();  // now DOM is updated
  }
</script>

Stores

Stores hold shared reactive state outside components. Any component can subscribe.

stores.js 
import { writable, readable, derived, get } from 'svelte/store';

// writable — can be set/updated externally
export const count = writable(0);
count.set(10);
count.update(n => n + 1);
const value = get(count);   // read without subscribing

// readable — external source, no external set
export const time = readable(new Date(), (set) => {
  const interval = setInterval(() => set(new Date()), 1000);
  return () => clearInterval(interval);  // cleanup
});

// derived — computed from other stores
export const doubled = derived(count, $count => $count * 2);
export const summary = derived(
  [count, time],
  ([$count, $time]) => `${$count} at ${$time.toLocaleTimeString()}`
);
Using stores in a component 
<script>
  import { count, doubled } from './stores.js';
</script>

<!-- $ prefix auto-subscribes and unsubscribes -->
<p>Count: {$count}</p>
<p>Doubled: {$doubled}</p>
<button on:click={() => $count++}>Increment</button>

Context

Context passes data through the component tree without props. Unlike stores, context is per-component-tree.

context.svelte 
// Parent.svelte
<script>
  import { setContext } from 'svelte';
  import { writable } from 'svelte/store';

  const theme = writable('dark');
  setContext('theme', theme);
</script>

// Child.svelte (any depth)
<script>
  import { getContext } from 'svelte';
  const theme = getContext('theme');
</script>

<div class={$theme}>Themed</div>

Transitions & Animations

transitions.svelte 
<script>
  import { fade, fly, slide, scale, blur } from 'svelte/transition';
  import { flip } from 'svelte/animate';
  import { tweened, spring } from 'svelte/motion';

  let visible = true;

  // Tweened store — smooth interpolation
  const progress = tweened(0, { duration: 400 });
  progress.set(0.75);   // animates to 75%

  // Spring — physics-based motion
  const coords = spring({ x: 0, y: 0 });
</script>

<!-- transition: — both enter and leave -->
{#if visible}
  <p transition:fade>Fades in and out</p>
{/if}

<!-- With options -->
<div transition:fly={{ y: 200, duration: 500 }}>Fly in</div>

<!-- in: / out: separately -->
<div in:slide out:scale>Different in/out</div>

<!-- animate:flip for list reordering -->
{#each items as item (item.id)}
  <div animate:flip={{ duration: 300 }}>{item.name}</div>
{/each}

<!-- Progress bar -->
<progress value={$progress} />

Actions

Actions are functions that attach behaviour to DOM elements. Reusable and side-effect focused.

actions.js 
// Action = function(node, params) => { update?, destroy? }
function clickOutside(node, handler) {
  const handleClick = (e) => {
    if (!node.contains(e.target)) handler();
  };
  document.addEventListener('click', handleClick, true);
  return {
    destroy() { document.removeEventListener('click', handleClick, true); }
  };
}

function tooltip(node, text) {
  // ... setup tooltip
  return {
    update(newText) { /* text changed */ },
    destroy() { /* cleanup */ }
  };
}

// Usage in template:
// <div use:clickOutside={close}>...</div>
// <button use:tooltip={"Click me"}>...</button>

SvelteKit

SvelteKit is the full-stack framework built on Svelte. It provides file-based routing, SSR, SSG, and API routes.

SvelteKit structure 
// src/routes/
//   +page.svelte            → route page component
//   +page.server.js         → server-side data loading
//   +layout.svelte          → shared layout
//   +layout.server.js       → layout data
//   +error.svelte           → error page
//   blog/[slug]/+page.svelte → dynamic route
//   api/users/+server.js    → API endpoint
+page.server.js — load data 
// Runs on the server, data is available to the page
export async function load({ params, fetch, cookies }) {
  const post = await fetch(`/api/posts/${params.slug}`)
    .then(r => r.json());

  if (!post) {
    throw error(404, 'Post not found');
  }

  return { post };
}
+page.svelte 
<script>
  // data comes from the load function above
  export let data;
</script>

<h1>{data.post.title}</h1>
<p>{data.post.content}</p>
+server.js — API route 
import { json } from '@sveltejs/kit';

export async function GET({ params }) {
  const user = await db.find(params.id);
  return json(user);
}

export async function POST({ request }) {
  const body = await request.json();
  const user = await db.create(body);
  return json(user, { status: 201 });
}
