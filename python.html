Python

A high-level, interpreted, dynamically typed language built for readability. Used everywhere from scripts to large-scale web apps, data science, and AI.

Interpreted

Executes line-by-line. No compile step.

Dynamic Typing

Types resolved at runtime, not declared upfront.

Indentation-based

Blocks use indentation, not curly braces.

Batteries Included

Rich standard library for most common tasks.

OOP & Classes

oop.py 
class Animal:
    count = 0   # class attribute

    def __init__(self, name, sound):
        self.name  = name   # instance attribute
        self.sound = sound
        Animal.count += 1

    def speak(self):        # instance method
        return f"{self.name} says {self.sound}"

    def __repr__(self):     # developer representation
        return f"Animal({self.name!r})"

    @classmethod
    def get_count(cls): return cls.count

    @staticmethod
    def validate(name): return bool(name)

    @property
    def info(self): return f"{self.name} ({self.sound})"


# Inheritance
class Dog(Animal):
    def __init__(self, name, breed):
        super().__init__(name, "woof")
        self.breed = breed

    def speak(self):         # override
        return super().speak() + "!"


# Dataclass — auto __init__, __repr__, __eq__
from dataclasses import dataclass

@dataclass
class Point:
    x: float
    y: float

    def distance(self):
        return (self.x**2 + self.y**2) ** 0.5

print(Point(3, 4).distance())   # 5.0

Lambdas & Closures

lambdas.py 
# Lambda — anonymous single-expression function
square = lambda x: x**2
add    = lambda a, b: a + b

# Use with map / filter / sorted
nums = [5, 2, 8, 1]
list(map(lambda x: x**2, nums))
list(filter(lambda x: x > 3, nums))
sorted(nums, key=lambda x: -x)

# Closure — inner function captures enclosing scope
def make_counter():
    count = 0
    def counter():
        nonlocal count
        count += 1
        return count
    return counter

c = make_counter()
print(c(), c(), c())   # 1 2 3

# functools.partial — pre-fill arguments
from functools import partial, reduce

double = partial(lambda a, b: a * b, 2)
print(double(5))   # 10

product = reduce(lambda a, b: a * b, [1, 2, 3, 4])  # 24

Decorators

decorators.py 
import functools, time

# Decorator = function that wraps another function
def timer(func):
    @functools.wraps(func)       # preserves metadata
    def wrapper(*args, **kwargs):
        t = time.perf_counter()
        result = func(*args, **kwargs)
        print(f"{func.__name__}: {time.perf_counter()-t:.4f}s")
        return result
    return wrapper

@timer
def slow(): time.sleep(0.1)

# Decorator factory (decorator with arguments)
def retry(times=3, exceptions=(Exception,)):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*a, **kw):
            for i in range(times):
                try: return func(*a, **kw)
                except exceptions:
                    if i == times - 1: raise
        return wrapper
    return decorator

@retry(times=3, exceptions=(IOError,))
def fetch(url): ...

# Class decorator
@timer
@retry(times=2)
def my_func(): ...   # decorators apply bottom-up

Generators

generators.py 
# Generator function — yields values lazily
def countdown(n):
    while n > 0:
        yield n
        n -= 1

for v in countdown(5): print(v)   # 5 4 3 2 1

# Infinite generator
def fibonacci():
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b

import itertools
first10 = list(itertools.islice(fibonacci(), 10))

# yield from — delegates to sub-generator
def chain(*iters):
    for it in iters:
        yield from it

# send() — two-way communication
def accumulator():
    total = 0
    while True:
        v = yield total
        if v is None: break
        total += v

acc = accumulator()
next(acc)           # prime
acc.send(10)        # 10
acc.send(20)        # 30

Context Managers

context.py 
# Class-based
class Timer:
    import time
    def __enter__(self):
        self._t = self.time.perf_counter()
        return self
    def __exit__(self, *args):
        print(f"{self.time.perf_counter()-self._t:.4f}s")

with Timer():
    sum(range(10**6))

# Generator-based (simpler)
from contextlib import contextmanager, suppress

@contextmanager
def db_connection(url):
    conn = connect(url)
    try:
        yield conn
    finally:
        conn.close()    # always runs

with db_connection("...") as db:
    db.execute("SELECT 1")

# Suppress specific exceptions
with suppress(FileNotFoundError):
    Path("missing.txt").unlink()

# Multiple context managers
with open("a") as a, open("b") as b:
    b.write(a.read())

Concurrency

Key insight: Python's GIL limits true CPU parallelism for threads. Use threading for I/O-bound work, multiprocessing for CPU-bound, and asyncio for high-concurrency I/O.
concurrency.py 
# asyncio — concurrent I/O without threads
import asyncio

async def fetch(url):
    await asyncio.sleep(1)   # simulate I/O
    return f"data from {url}"

async def main():
    urls = ["a.com", "b.com", "c.com"]
    results = await asyncio.gather(*[fetch(u) for u in urls])
    print(results)

asyncio.run(main())

# Threading — for I/O-bound tasks
import threading

def worker(n):
    print(f"Thread {n}")

threads = [threading.Thread(target=worker, args=(i,)) for i in range(5)]
for t in threads: t.start()
for t in threads: t.join()

# ProcessPoolExecutor — CPU-bound parallelism
from concurrent.futures import ProcessPoolExecutor

def cpu_task(n): return n ** 2

with ProcessPoolExecutor() as pool:
    results = list(pool.map(cpu_task, range(100)))
